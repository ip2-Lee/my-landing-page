        #mobile-frame-container {

            display: none; /* 자바스크립트가 PC라고 판단하기 전까지 숨김 */

    <script type="text/javascript">

        // ▼▼▼ 여기에 본인의 윅스 모바일 페이지 주소를 넣으세요 ▼▼▼

        var destinationUrl = "https://www.mysite.com"; 

        // ▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲


        // 정밀 기기 판별 함수

        function detectDevice() {

            var ua = navigator.userAgent || navigator.vendor || window.opera;

            

            // 1. 모바일 OS 정규식 체크 (Android, iOS 등)

            var isMobileOS = /android|avantgo|blackberry|blazer|compal|elaine|fennec|hiptop|iemobile|ip(hone|od|ad)|iris|kindle|lge |maemo|midp|mmp|mobile|o2|opera m(ob|in)i|palm( os)?|p(ixi|re)\/|plucker|pocket|psp|smartphone|symbian|treo|up\.(browser|link)|vodafone|wap|windows ce|xda|xiino/i.test(ua);

            

            // 2. 터치 포인트 체크 (아이패드 등 태블릿 대응)

            var isTouch = (navigator.maxTouchPoints && navigator.maxTouchPoints > 2);


            // 최종 판단: 모바일 OS이거나, 터치 기반이면서 화면이 좁은 경우

            // (화면 너비 768px 이하는 모바일로 간주)

            var isMobileView = isMobileOS || (isTouch && window.innerWidth <= 768);


            return isMobileView;

        }


        // 실행 로직

        window.onload = function() {

            var isMobile = detectDevice();


            if (isMobile) {

                // [모바일 접속] -> 즉시 리다이렉트

                document.getElementById('loading').style.display = 'block';

                document.getElementById('loading').innerText = '모바일 페이지로 이동 중...';

                

                // 히스토리를 남기지 않고 교체 (뒤로가기 시 루프 방지)

                window.location.replace(destinationUrl);

            } else {

                // [PC 접속] -> iframe 컨테이너 활성화

                var container = document.getElementById('mobile-frame-container');

                var frame = document.getElementById('content-frame');

                

                container.style.display = 'block'; // 박스 보이기

                frame.src = destinationUrl; // iframe에 주소 주입

            }

        };

    </script>
